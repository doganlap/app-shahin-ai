CREATE TABLE RoleProfiles (
    Id UNIQUEIDENTIFIER PRIMARY KEY,
    RoleCode NVARCHAR(50) NOT NULL,      -- 'CO', 'RM', 'AM'

    Layer NVARCHAR(50) NOT NULL,          -- 'Executive', 'Management', 'Operational'

    ApprovalLevel INT,

    ParticipatingWorkflows NVARCHAR(MAX), -- 'WF-NCA-ECC-001,WF-SAMA-CSF-001'

    Steps NVARCHAR(MAX),                  -- JSON array of steps

    IsActive BIT DEFAULT 1

    Status NVARCHAR(50) DEFAULT 'Pending',
    ...
